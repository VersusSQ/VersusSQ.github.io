计算已经出生多久和距下一次生日的天数 

import time

date = {1: 31, 2: 28, 3: 31, 4: 30, 5: 31, 6: 30,
        7: 31, 8: 31, 9: 30, 10: 31, 11: 30, 12: 31}
month = {"Jan": 1, "Feb": 2, "Mar": 3, "Apr": 4, "May": 5, "Jun": 6,
         "Jul": 7, "Aus": 8, "Sep": 9, "Oct": 10, "Nov": 11, "Dec": 12}

# 计算出生了多久的函数
def live_days(start_date, end_date):
    start = {}

            print("从出生至今已%d天" % sum_day)

# 计算距下次过生日的天数
def next_birthday(start_date, end_date):
    start = {}
    end = {}
    days = 0
    start["yy"], start["mm"], start["dd"] = int(start_date[0:4]), int(start_date[4:6]), int(start_date[6:])
    end["yy"], end["mm"], end["dd"] = int(end_date[4]), month[end_date[1]], int(end_date[2])
    date[2] = 28

    # 距离下次过生日的天数
    # 当前日期还未到生日那天
    if end["mm"] < start["mm"]:
        days = 0
        if end["yy"] % 400 == 0 or (end["yy"] % 4 == 0 and end["yy"] % 100 != 0):
            date[2] = 29
        for i in range(end["mm"]+1, start["mm"]):
            days += date[i]
        sum_day = days + start["dd"] + (date[end["mm"]] - end["dd"])
        print("距离下次过生日还有%d天" % sum_day)

    elif end["mm"] == start["mm"] and end["dd"] <= start["dd"]:
        print("距离下次过生日还有%d天" % (start["dd"] - end["dd"]))

    elif end["mm"] > start["mm"]:
        # end_date
        end_days = 0
        if (end["yy"] + 1) % 400 == 0 or ((end["yy"] + 1) % 4 == 0 and (end["yy"] + 1) % 100 != 0):
            date[2] = 29

        if start["mm"] == 1:
            result_end = start["dd"]
        else:
            for i in range(1, start["mm"]):
                end_days += date[i]
            result_end = start["dd"] + end_days

        date[2] = 28
        # start_date
        start_days = 0
        if end["yy"] % 400 == 0 or (end["yy"] % 4 == 0 and end["yy"] % 100 != 0):
            date[2] = 29
        for i in range(end["mm"] + 1, 13):
            start_days += date[i]
        result_start = date[end["mm"]] - end["dd"] + start_days
        sum_day = result_end + result_start
        print(result_start)
        print(result_end)
        print("距离下次过生日还有%d天" % sum_day)


def main():
    while True:
        birth_day = input("请输入你的出生年月日（格式：20170112）：")
        now = time.ctime().split()
        live_days(birth_day, now)
        next_birthday(birth_day, now)

if __name__ == "__main__":
    main()
